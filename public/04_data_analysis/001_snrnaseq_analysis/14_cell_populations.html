sce$fraction_disesase_doner <-
  paste0(sce$prep, "_", sce$diagnosis, "_", sce$donor)

  # convert to dataframe and rename columns

  )

# Get unique cell types
unique_cell_types <- unique(Idents(sce))

# Create a function to perform the tests and adjust p-values
perform_tests <- function(cell_type) {
  # Subset cell counts for the current cell type in AD and control groups
  cell_counts_AD <- sce$cell_counts[sce$group == "AD" & sce$cell_type == cell_type]
  cell_counts_control <- sce$cell_counts[sce$group == "control" & sce$cell_type == cell_type]
  
  # Perform t-test
  ttest_result <- t.test(cell_counts_AD, cell_counts_control)
  ttest_pvalue <- ttest_result$p.value
  
  # Perform Wilcoxon rank-sum test
  wilcox_result <- wilcox.test(cell_counts_AD, cell_counts_control)
  wilcox_pvalue <- wilcox_result$p.value
  
  # Return adjusted p-values
  return(list(CellType = cell_type, TTest_PValue = ttest_pvalue, Wilcox_PValue = wilcox_pvalue))
}

# Apply the function to each unique cell type using map
test_results <- map(unique_cell_types, perform_tests)

# Extract p-values
ttest_pvalues <- map_dbl(test_results, "TTest_PValue")
wilcox_pvalues <- map_dbl(test_results, "Wilcox_PValue")

# Adjust p-values using the Benjamini-Hochberg procedure
ttest_pvalues_adjusted <- p.adjust(ttest_pvalues, method = "BH")
wilcox_pvalues_adjusted <- p.adjust(wilcox_pvalues, method = "BH")

# Create a dataframe with the results
result_df <- data.frame(CellType = unique_cell_types, TTest_PValue = ttest_pvalues, TTest_PValue_Adjusted = ttest_pvalues_adjusted, Wilcox_PValue = wilcox_pvalues, Wilcox_PValue_Adjusted = wilcox_pvalues_adjusted)

#| eval: false
# Get unique cell types
unique_cell_types <- unique(Idents(sce))

# Create a function to perform the tests and adjust p-values
perform_tests <- function(cell_type) {
  # Subset cell counts for the current cell type in AD and control groups
  cell_counts_AD <- sce$cell_counts[sce$group == "AD" & sce$cell_type == cell_type]
  cell_counts_control <- sce$cell_counts[sce$group == "control" & sce$cell_type == cell_type]
  
  # Perform t-test
  ttest_result <- t.test(cell_counts_AD, cell_counts_control)
  ttest_pvalue <- ttest_result$p.value
  
  # Perform Wilcoxon rank-sum test
  wilcox_result <- wilcox.test(cell_counts_AD, cell_counts_control)
  wilcox_pvalue <- wilcox_result$p.value
  
  # Return adjusted p-values
  return(list(CellType = cell_type, TTest_PValue = ttest_pvalue, Wilcox_PValue = wilcox_pvalue))
}

# Apply the function to each unique cell type using map
test_results <- map(unique_cell_types, perform_tests)

# Extract p-values
ttest_pvalues <- map_dbl(test_results, "TTest_PValue")
wilcox_pvalues <- map_dbl(test_results, "Wilcox_PValue")

# Adjust p-values using the Benjamini-Hochberg procedure
ttest_pvalues_adjusted <- p.adjust(ttest_pvalues, method = "BH")
wilcox_pvalues_adjusted <- p.adjust(wilcox_pvalues, method = "BH")

# Create a dataframe with the results
result_df <- data.frame(CellType = unique_cell_types, TTest_PValue = ttest_pvalues, TTest_PValue_Adjusted = ttest_pvalues_adjusted, Wilcox_PValue = wilcox_pvalues, Wilcox_PValue_Adjusted = wilcox_pvalues_adjusted)
